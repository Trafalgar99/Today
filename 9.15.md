```js
//格式化时间
function dateFormat(date,format = "YYYY-MM-DD HH:mm:ss"){
    const config = {
        YYYY:date.getFullYear(),
		MM:date.getMonth()+1,
    	DD:date.getDate(),
    	HH:date.getHour(),
    	mm:date.getMinutes(),
    	ss:date.getSecond(),
    }
    for (const key in config){
        format = format.replace(key,config[key]);
    }
    return format;
}
```
