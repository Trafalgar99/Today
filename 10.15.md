## Vue

```html

<script>
	var news = {
        template:"news",
        props:{

                required:true//必须传递这个参数
```

2. 子组件呼叫父组件

```html
    <div id="hd">
      <news :lists="goods" @sum="total"></news> //自定义sum事件，触发sum事件时调用父组件的total方法
      total:{{totalPrice}}
    </div>
    <script>
      var news = {
        template: `
        <table border="1" width="90%">
        <tr>
          <th>商品名称</th>
          <th>商品价格</th>
          <th>商品数量</th>
        </tr>
        <tr v-for="(v,k) in lists">
          <td>{{v.title}}</td>
          <td>{{v.price}}</td>
          <td><input type="text" v-model="v.num" @keyup="sum"/></td> //当触发keyup时调用子组件的sum函数
        </tr>
      </table>
        `, //组件只能有一个根元素
        //配置组件数据
        data: function () {
          return {};
        },
        methods: {
          sum() {	//sum函数的功能是触发父组件的“sum”事件，这也是$emit方法的功能
            this.$emit("sum");
          },
        },
        props: ["lists", "f"],
      };
      new Vue({
        el: "#hd",
        components: { news },
        data: {
          totalPrice: 0,
          goods: [
            { title: "apple", price: 100, num: 1 },
            { title: "banana", price: 200, num: 1 },
            { title: "peach", price: 211, num: 1 },
          ],
        },
          //挂载点
        mounted() { //在vue渲染结束时会调用mounted方法
          this.total();
        },
        methods: {
          total() {
            this.totalPrice = 0;
            this.goods.forEach((element) => {
              this.totalPrice += element.price * element.num;
            });
          },
        },
      });
    </script>
```

3. 实现上一个功能的另外一种方法 --- sync

```html
    <div id="hd">
      <news :lists.sync="goods"></news> //将子组件和父组件的lists绑定，一边有变化，两边都会有感应
      total:{{totalPrice}}
    </div>
    <script>
      var news = {
        template: `
        <table border="1" width="90%">
        <tr>
          <th>商品名称</th>
          <th>商品价格</th>
          <th>商品数量</th>
        </tr>
        <tr v-for="(v,k) in lists">
          <td>{{v.title}}</td>
          <td>{{v.price}}</td>
          <td><input type="text" v-model="v.num"/></td>
        </tr>
      </table>
        `, //组件只能有一个根元素
        //配置组件数据
        data: function () {
          return {};
        },
        props: ["lists"],
      };
      new Vue({
        el: "#hd",
        components: { news },
        data: {
          goods: [
            { title: "apple", price: 100, num: 1 },
            { title: "banana", price: 200, num: 1 },
            { title: "peach", price: 211, num: 1 },
          ],
        },
        computed: { //使用计算属性代替函数，计算属性每当goods有变化都会重新计算
          totalPrice() {
            let sum = 0;
            this.goods.forEach((element) => {
              sum += element.price * element.num;
            });
            return sum;
          },
        },
      });
    </script>
```

5. 父组件设置子组件的模板数据

```html
    <div id="hd">
      <hd-list :data="news">
        <template v-slot="v"> // 必须在template内用v-slot属性接收子组件的参数
          <li>
            <h1>{{v.field.title}}</h1>
          </li>
        </template>
      </hd-list>
    </div>

    <script type="text/x-template" id="hdList">
      <ul>
        <slot v-for='v in data' :field='v'> //用:field传递子组件中的参数
        </slot>
      </ul>
    </script>
    <script>
      var hdList = {
        template: "#hdList", //组件只能有一个根元素
        //配置组件数据
        data: function () {
          return {};
        },
        props: ["data"],
      };
      new Vue({
        el: "#hd",
        components: { hdList },
        data: {
          news: [{ title: "xx" }, { title: "yy" }],
        },
      });
    </script>
```
