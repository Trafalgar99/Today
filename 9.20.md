10. 对象的复制

```js
//浅拷贝
let obj = Object.assign({}, hd);
let obj = {...hd};
//深拷贝
let obj = {
    name:"xxx",
    user:{
        name:"yyy"
    },
    arr:[1,2,3]
};
//递归算法实现深拷贝
function copy(obj){
	let res = obj instanceof Array?[]:{};	// typeof无法区分数组和对象
    for (const [k, v] of Object.entries(obj)){	//Object.entries()不论是数组还是对象都返回 键-值 结构
        res[k] = typeof v === 'object'?copy(v):v;
    }
    return res;
}
```

    value:'yyy',

21. proxy代理控制对象

```js
const hd = {name:"x"};
const proxy = new Proxy(hd,{
    get(obj,property){
        return obj[property];
    },
    set(obj,property,value){
        obj[property] = value;
        return true; //符合严格模式的要求
    }
});
proxy.name = "x";
console.log(proxy.name);
```

22. proxy代理控制函数

```js
function factorial(num){
    return num==1?1:num*factorial(num-1);
}
let proxy = new Proxy(factrial,{
    apply(func,obj,args){
        console.time('run');
        func.apply(this,args);
        console.timeEnd('run');
    },
});

proxy.apply({},[5]);	//会显示运行时间
```
